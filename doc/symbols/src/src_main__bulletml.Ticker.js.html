 16 bulletml.TickerFactory.prototype.create = function(config, action) {
 17     var topLabels = this._bulletml.getTopActionLabels();
 18     if (!action && topLabels.length > 0) {
 19         var parentTicker = new bulletml.Ticker();
 20         topLabels.forEach(function(label) {
 21             parentTicker.addSubTicker(this._create(config, label));
 22         });
 23         return parentTicker;
 24     } else {
 25         return this._create(config, action);
 26     }
 27 };
