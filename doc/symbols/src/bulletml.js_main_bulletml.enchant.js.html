 34     // BulletML(*.xml)をpreloadで読み込めるようにする.
 35     enchant.Game._loadFuncs["xml"] = function(src, callback) {
 36         var game = this;
 37         var xhr = new XMLHttpRequest();
 38         xhr.onreadystatechange = function(e) {
 39             if (xhr.readyState === 4) {
 40                 if (xhr.status !== 200 && xhr.status !== 0) {
 41                     throw new Error(xhr.status + ': '
 42                             + 'Cannot load an asset: ' + src);
 43                 }
 44 
 45                 if (xhr.responseXML != null) {
 46                     game.assets[src] = BulletML.build(xhr.responseXML);
 47                     callback();
 48                 } else if (xhr.responseText != null) {
 49                     game.assets[src] = BulletML.build(xhr.responseText);
 50                     callback();
 51                 } else {
 52                     throw new Error(xhr.status + ': '
 53                             + 'Cannot load an asset: ' + src);
 54                 }
 55             }
 56         };
 57 
 58         xhr.open('GET', src, true);
 59         if (xhr.overrideMimeType) {
 60             xhr.overrideMimeType("application/xml");
 61         }
 62         xhr.send(null);
 63     };

195             this.bulletml = bulletml;
196             this.seq = bulletml.sequence();
197             this.config = {
198                 width : 8,
199                 height : 8,
200                 image : enchant.Surface.load(enchant.bulletml.defaultImage),
201                 frame : 0,
202                 removeOnScreenOut : true,
203                 target : null,
204                 onfire : function() {
205                 },
206                 onenterframe : function() {
207                 },
208                 onremove : function() {
209                 }
210             };
211             if (config) {
212                 for ( var prop in config) {
213                     if (config.hasOwnProperty(prop)) {
214                         this.config[prop] = config[prop];
215                     }
216                 }
217             }
218 
219             this.age = 0;
220             this.cursor = 0;
221             this.waitTo = -1;
222             this.lastDirection = 0;
223             this.lastSpeed = 1;
224             this._attacker = null;
225             this.completed = false;
226         },
227         /**
228          * 攻撃を行う敵機.
229          * 
230          * @type {enchant.Node}
231          */

240         /**
241          * フレームごとに呼び出される.
242          */
243         tick : function() {
244             if (this.age++ < this.waitTo || this.completed) {
245                 return;
246             }
247 
248             for ( var end = this.seq.length; this.cursor < end; this.cursor++) {
249                 var command = this.seq[this.cursor];
250                 switch (command.commandName) {
251                 case "fire":
252                     enchant.bulletml.fireBullet.call(this, command,
253                             this._attacker.scene, this.config, this._attacker);
254                     break;
255                 case "wait":
256                     this.waitTo = this.age + eval(command.value);
257                     this.cursor += 1;
258                     return;
259                 case "loopEnd":
260                     if (command.loopCount == -1) {
261                         command.loopCount = 0;
262                     } else {
263                         command.loopCount += 1;
264                     }
265                     if (command.loopCount < command.times - 1) {
266                         while (0 < this.cursor
267                                 && this.seq[this.cursor] != command.start) {
268                             this.cursor -= 1;
269                         }
270                     } else {
271                         command.loopCount = -1;
272                     }
273                     break;
274                 }
275             }
276             this._attacker.dispatchEvent(new Event("completeAttack"));
277             this.completed = true;
278         },

299     /**
300      * 弾を発射する.
301      * 
302      * @param {BulletML.Fire}
303      *            fireCmd 発射コマンド
304      * @param {enchant.Group}
305      *            scene 弾スプライトの親となるノード
306      * @param {Object}
307      *            config 設定
308      * @param {enchant.Node}
309      *            attacker 弾を発射するオブジェクト
310      */
311     enchant.bulletml.fireBullet = function(fireCmd, scene, config, attacker) {
312         // console.log(fireCmd.direction.type, fireCmd.direction.value);
313         var pattern;
314         if (this instanceof enchant.bulletml.AttackPattern) {
315             pattern = this;
316         } else if (this.pattern) {
317             pattern = this.pattern;
318         }
319 
320         var w = config.width;
321         var h = config.height;
322         var x = attacker.x + ((attacker.width || 0) - w) / 2;
323         var y = attacker.y + ((attacker.height || 0) - h) / 2;
324         var b = new enchant.bulletml.Bullet(x, y, w, h, pattern, fireCmd.bullet);
325 
326         b.image = config.image;
327         b.frame = config.frame;
328 
329         if (fireCmd.direction) {
330             var dv = toRadian(eval(fireCmd.direction.value));
331             switch (fireCmd.direction.type) {
332             case "aim":
333                 if (config.target) {
334                     var t = config.target;
335                     b.direction = radiusAtoB(b, t) + dv;
336                 } else {
337                     b.direction = dv;
338                 }
339                 break;
340             case "absolute":
341                 b.direction = dv - Math.PI / 2; // 真上が0度
342                 break;
343             case "relative":
344                 if (this.direction) {
345                     b.direction = this.direction + dv;
346                 } else if (this.lastDirection) {
347                     b.direction = this.lastDirection + dv;
348                 }
349                 break;
350             case "sequence":
351                 b.direction = this.lastDirection + dv;
352                 break;
353             }
354         }
355 
356         this.lastDirection = b.direction;
357 
358         if (fireCmd.speed) {
359             var sv = eval(fireCmd.speed.value);
360             switch (fireCmd.speed.type) {
361             case "relative":
362             case "sequence":
363                 b.speed = this.lastSpeed + sv;
364                 break;
365             case "absolute":
366             default:
367                 b.speed = sv;
368                 break;
369             }
370         }
371         this.lastSpeed = b.speed;
372 
373         b.seq = fireCmd.bullet.sequence();
374 
375         if (scene) {
376             scene.addChild(b);
377         }
378 
379         // console.log(b.direction);
380 
381         pattern.config.onfire.call(b);
382         return b;
383     };

389      * @scope enchant.bulletml.Bullet.prototype

392     enchant.bulletml.Bullet = enchant.Class.create(enchant.Sprite, {
393         /**
394          * @param {number}
395          *            x 初期位置x
396          * @param {number}
397          *            y 初期位置y
398          * @param {number}
399          *            width スプライトの幅
400          * @param {number}
401          *            height スプライトの高さ
402          * @param {enchant.bulletml.AttackPattern}
403          *            attackPattern 攻撃パターン
404          * @param {BulletML.Bullet}
405          *            bulletSpec 弾
406          * @constructs
407          */
408         initialize : function(x, y, width, height, attackPattern, bulletSpec) {
409             enchant.Sprite.call(this, width, height);
410             this.x = x;
411             this.y = y;
412             this.pattern = attackPattern;
413             this.cursor = 0;
414             this.waitTo = -1;
415             this.lastDirection = 0;
416             this.lastSpeed = 1;
417 
418             if (bulletSpec.direction) {
419                 var dv = toRadian(eval(bulletSpec.direction.value));
420                 switch (bulletSpec.direction.type) {
421                 case "absolute":
422                     this.direction = dv - Math.PI / 2; // 真上が0度
423                     break;
424                 case "sequence":
425                     this.direction = attackPattern.lastDirection + dv;
426                     break;
427                 case "aim":
428                     if (attackPattern.config.target) {
429                         var a = attackPattern.attacker;
430                         var t = attackPattern.config.target;
431                         this.direction = radiusAtoB(a, t) + dv;
432                     } else {
433                         this.direction = dv;
434                     }
435                     break;
436                 }
437             }
438 
439             if (bulletSpec.speed) {
440                 var sv = eval(bulletSpec.speed.value);
441                 switch (bulletSpec.speed.type) {
442                 case "absolute":
443                 case "relative":
444                     this.speed = sv;
445                     break;
446                 case "sequence":
447                     this.speed = attackPattern.lastSpeed + sv;
448                     break;
449                 }
450             }
451             attackPattern.lastSpeed = this.speed;
452 
453             this.accelH = 0;
454             this.accelV = 0;
455 
456             this._changeDirection = null;
457             this._changeSpeed = null;
458 
459             this.scw = enchant.Game.instance.width;
460             this.sch = enchant.Game.instance.height;
461 
462             this.addEventListener("enterframe", this.tick);
463         },
464         /**
465          * 
466          */
467         tick : function() {
468             this.pattern.config.onenterframe.call(this);
469 
470             this._changeDirection && this._changeDirection();
471             this._changeSpeed && this._changeSpeed();
472 
473             this.x += Math.cos(this.direction) * this.speed * 2;
474             this.y += Math.sin(this.direction) * this.speed * 2;
475             if (this.rotation != undefined) {
476                 this.rotation = this.direction;
477             }
478 
479             if (this.pattern.config.removeOnScreenOut) {
480                 if (this.x < 0 || this.scw + this.width < this.x || this.y < 0
481                         || this.sch + this.height < this.y) {
482                     this.pattern.config.onremove.call(this);
483                     this.parentNode.removeChild(this);
484                 }
485             }
486 
487             if (this.age < this.waitTo) {
488                 return;
489             }
490 
491             for ( var end = this.seq.length; this.cursor < end; this.cursor++) {
492                 var command = this.seq[this.cursor];
493                 switch (command.commandName) {
494                 case "fire":
495                     enchant.bulletml.fireBullet.call(this, command, this.scene,
496                             this.pattern.config, this);
497                     break;
498                 case "wait":
499                     this.waitTo = this.age + eval(command.value);
500                     this.cursor += 1;
501                     return;
502                 case "loopEnd":
503                     if (command.loopCount == -1) {
504                         command.loopCount = 0;
505                     } else {
506                         command.loopCount += 1;
507                     }
508                     if (command.loopCount < command.times - 1) {
509                         while (0 < this.cursor
510                                 && this.seq[this.cursor] != command.start) {
511                             this.cursor -= 1;
512                         }
513                     } else {
514                         command.loopCount = -1;
515                     }
516                     break;
517                 case "changeDirection":
518                     this.changeDirection(command);
519                     break;
520                 case "changeSpeed":
521                     this.changeSpeed(command);
522                     break;
523                 case "accel":
524                     // TODO
525                     break;
526                 case "vanish":
527                     this.pattern.config.onremove.call(this);
528                     if (this.parentNode) {
529                         this.parentNode.removeChild(this);
530                     }
531                     return;
532                 }
533             }
534         },
535         /**
536          * @param {BulletML.ChangeDirection}
537          *            cmd
538          */
539         changeDirection : function(cmd) {
540             var incr;
541             var finalVal;
542             var endAge;
543 
544             var d = eval(cmd.direction.value);
545             var t = eval(cmd.term);
546             switch (cmd.direction.type) {
547             case "aim":
548                 var tar = this.pattern.config.target;
549                 if (!tar) {
550                     return;
551                 }
552                 finalVal = radiusAtoB(this, tar) + toRadian(d);
553                 incr = rel(finalVal - this.direction) / t;
554                 break;
555             case "absolute":
556                 finalVal = toRadian(d) - Math.PI / 2;
557                 incr = rel(finalVal - this.direction) / t;
558                 break;
559             case "relative":
560                 finalVal = this.direction + toRadian(d);
561                 incr = rel(finalVal - this.direction) / t;
562                 break;
563             case "sequence":
564                 incr = toRadian(d);
565                 finalVal = this.direction + incr * t;
566                 break;
567             }
568             endAge = this.age + t;
569 
570             this._changeDirection = function() {
571                 this.direction += incr;
572                 if (this.age == endAge) {
573                     this.direction = finalVal;
574                     this._changeDirection = null;
575                 }
576             };
577         },
578         /**
579          * @param {BulletML.ChangeSpeed}
580          *            cmd
581          */
582         changeSpeed : function(cmd) {
583             var incr;
584             var finalVal;
585             var endAge;
586 
587             var s = eval(cmd.speed.value);
588             var t = eval(cmd.term);
589             switch (cmd.speed.type) {
590             case "absolute":
591                 finalVal = s;
592                 incr = (finalVal - this.speed) / t;
593                 break;
594             case "relative":
595                 finalVal = s + this.speed;
596                 incr = (finalVal - this.speed) / t;
597                 break;
598             case "sequence":
599                 incr = s;
600                 finalVal = this.speed + incr * t;
601                 break;
602             }
603             endAge = this.age + t;
604 
605             this._changeSpeed = function() {
606                 this.speed += incr;
607                 if (this.age == endAge) {
608                     this.speed = finalVal;
609                     this._changeSpeed = null;
610                 }
611             }
612         }
613     });

636     /**
637      * ラジアンを -π<= rad < π の範囲に正規化する.
638      */
639     function rel(radian) {
640         while (radian <= -Math.PI) {
641             radian += Math.PI * 2;
642         }
643         while (Math.PI < radian) {
644             radian -= Math.PI * 2;
645         }
646         return radian;
647     }
648     /**
649      * スプライトAから見たスプライトBの方向をラジアンで返す.
650      * 
651      * @param {enchant.Node}
652      *            a スプライトA
653      * @param {enchant.Node}
654      *            b スプライトB
655      */
656     function radiusAtoB(a, b) {
657         var ca = {
658             x : a.x + (a.width || 0) / 2,
659             y : a.y + (a.height || 0) / 2
660         };
661         var cb = {
662             x : b.x + (b.width || 0) / 2,
663             y : b.y + (b.height || 0) / 2
664         };
665         return Math.atan2(cb.y - ca.y, cb.x - ca.x);
666     }
