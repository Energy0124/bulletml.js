 32     // BulletML(*.bml, *.xml)をpreloadで読み込めるようにする.
 33     enchant.Game._loadFuncs["bml"] = enchant.Game._loadFuncs["xml"] = function(
 34             src, callback) {
 35         var game = this;
 36         var xhr = new XMLHttpRequest();
 37         xhr.onreadystatechange = function(e) {
 38             if (xhr.readyState === 4) {
 39                 if (xhr.status !== 200 && xhr.status !== 0) {
 40                     throw new Error(xhr.status + ': '
 41                             + 'Cannot load an asset: ' + src);
 42                 }
 43 
 44                 if (xhr.responseXML != null) {
 45                     var bulletml = BulletML.build(xhr.responseXML);
 46                     if (bulletml) {
 47                         game.assets[src] = new enchant.bulletml.AttackPattern(
 48                                 bulletml);
 49                     } else {
 50                         console.warn(src + "は妥当なBulletMLではありません。");
 51                         game.assets[src] = xhr.responseXML;
 52                     }
 53                     callback();
 54                 } else if (xhr.responseText != null) {
 55                     game.assets[src] = BulletML.build(xhr.responseText);
 56                     callback();
 57                 } else {
 58                     throw new Error(xhr.status + ': '
 59                             + 'Cannot load an asset: ' + src);
 60                 }
 61             }
 62         };
 63 
 64         xhr.open('GET', src, true);
 65         if (xhr.overrideMimeType) {
 66             xhr.overrideMimeType("application/xml");
 67         }
 68         xhr.send(null);
 69     };

 80     enchant.EventTarget.prototype.removeDanmaku = function() {
 81         if (!this._listeners["enterframe"]
 82                 || this._listeners["enterframe"].length === 0) {
 83             return;
 84         }
 85         var remove = [];
 86         for ( var i = this._listeners["enterframe"].length; i--;) {
 87             if (this._listeners["enterframe"][i].isDanmaku) {
 88                 remove[remove.length] = this._listeners["enterframe"][i];
 89             }
 90         }
 91         for ( var i = remove.length; i--;) {
 92             this.removeEventListener("enterframe", remove[i]);
 93         }

197         /**

199          * <br>

201          *

205          *            <table border=1>
206          *            <tr>
207          *            <th>プロパティ名</th>
208          *            <th>型</th>
209          *            <th>設定内容</th>
210          *            <th>デフォルト値</th>
211          *            <th>必須</th>
212          *            </tr>
213          *            <tr>
214          *            <td>target</td>
215          *            <td>enchant.Node</td>
216          *            <td>攻撃の標的となるオブジェクト</td>
217          *            <td>null</td>
218          *            <td>○</td>
219          *            </tr>
220          *            <tr>
221          *            <td>addTarget</td>
222          *            <td>enchant.Group</td>
223          *            <td>生成した弾を追加するノード</td>
224          *            <td>攻撃を実行するノードのparentNode</td>
225          *            <td></td>
226          *            </tr>
227          *            <tr>
228          *            <td>bulletFactory</td>
229          *            <td>function</td>
230          *            <td>弾ノードを生成する関数</td>
231          *            <td>小さな赤い弾を生成</td>
232          *            <td></td>
233          *            </tr>
234          *            <tr>
235          *            <td>updateProperties</td>
236          *            <td>boolean</td>
237          *            <td>弾のプロパティ(direction, speed)を更新するかどうか</td>
238          *            <td>false</td>
239          *            <td></td>
240          *            </tr>

243          *            <td>function</td>
244          *            <td>弾が画面内に存在することを判定する関数</td>
245          *            <td>Gameインスタンスの大きさをベースにして判定する</td>
246          *            <td></td>
247          *            </tr>
248          *            <tr>
249          *            <td>rank</td>
250          *            <td>number</td>
251          *            <td>弾幕ランク.BulletMLの$rankに対応する.0.0～1.0の範囲で指定.</td>
252          *            <td>0</td>
253          *            <td></td>
254          *            </tr>
255          *            <tr>
256          *            <td>speedRate</td>
257          *            <td>number</td>
258          *            <td>弾速度の補正倍率.</td>
259          *            <td>1</td>
260          *            <td></td>
261          *            </tr>
262          *            </table>
263          *            設定する項目がtargetのみの場合、標的オブジェクトを直接引数として渡すことが可能.
264          * @param {string}
265          *            [action] 最初に読み込むactionのラベル.<br>
266          *            省略可.
267          */
268         createTicker : function(config, action) {
269             var topLabels = this._bulletml.getTopActionLabels()
270             if (action === undefined && topLabels.length > 1) {
271                 // top***対応.
272                 // actionラベルtop***が定義されていた場合、それらを同時に動かす.
273                 var tickers = [];
274                 for ( var i = 0, end = topLabels.length; i < end; i++) {
275                     tickers[tickers.length] = this._createTicker(config, topLabels[i]);
276                 }
277                 var parentTicker = function() {
278                     if (parentTicker.complete) {
279                         return;
280                     }
281                     for ( var i = tickers.length; i--;) {
282                         tickers[i].call(this);
283                     }
284                     if (parentTicker.compChildCount == tickers.length) {
285                         parentTicker.complete = true;
286                         this.dispatchEvent(new Event("completeAttack"));
287                     }
288                 };
289                 for ( var i = tickers.length; i--;) {
290                     tickers[i].parentTicker = parentTicker;
291                 }
292 
293                 parentTicker.compChildCount = 0;
294                 parentTicker.completeChild = function() {
295                     this.compChildCount++;
296                 };
297 
298                 parentTicker.compChildCount = 0;
299                 parentTicker.complete = false;
300 
301                 parentTicker.isDanmaku = true;
302 
303                 return parentTicker;
304             } else {
305                 return this._createTicker(config, action);
306             }
307         },
308         _createTicker : function(config, action) {
309             config = this._getConf(config);
310             if (!config.target) {
311                 throw new Error("target is undefined in config.");
312             }
313 
314             // var ticker = tickerPool.get();
315             var ticker = function() {
316                 if (!ticker._pattern) {
317                     return;
318                 }
319 
320                 var conf = ticker.config;
321 
322                 // update direction
323                 if (this.age < ticker.chDirEnd) {
324                     ticker.direction += ticker.dirIncr;
325                 } else if (this.age == ticker.chDirEnd) {
326                     ticker.direction = ticker.dirFin;
327                 }
328 
329                 // update speed
330                 if (this.age < ticker.chSpdEnd) {
331                     ticker.speed += ticker.spdIncr;
332                 } else if (this.age == ticker.chSpdEnd) {
333                     ticker.speed = ticker.spdFin;
334                 }
335 
336                 // update accel
337                 if (this.age < ticker.aclEnd) {
338                     ticker.speedH += ticker.aclIncrH;
339                     ticker.speedV += ticker.aclIncrV;
340                 } else if (this.age == ticker.aclEnd) {
341                     ticker.speedH = ticker.aclFinH;
342                     ticker.speedV = ticker.aclFinV;
343                 }
344 
345                 // move sprite
346                 this.x += Math.cos(ticker.direction) * ticker.speed
347                         * conf.speedRate;
348                 this.y += Math.sin(ticker.direction) * ticker.speed
349                         * conf.speedRate;
350                 this.x += ticker.speedH * conf.speedRate;
351                 this.y += ticker.speedV * conf.speedRate;
352 
353                 // test out of world
354                 if (!conf.testInWorld(this)) {
355                     this.parentNode.removeChild(this);
356                     ticker.completed = true;
357                     if (ticker.parentTicker) {
358                         ticker.parentTicker.completeChild();
359                     } else {
360                         this.dispatchEvent(new Event("completeAttack"));
361                     }
362                     return;
363                 }
364 
365                 // set direction, speed to bullet
366                 if (conf.updateProperties) {
367                     this.direction = toDegree(ticker.direction
368                             + Math.PI / 2);
369                     this.speed = ticker.speed;
370                 }
371 
372                 // proccess walker
373                 if (this.age < ticker.waitTo || ticker.completed) {
374                     return;
375                 }
376                 var cmd;
377                 while (cmd = ticker.walker.next()) {
378                     switch (cmd.commandName) {
379                     case "fire":
380                         ticker._pattern._fire.call(this, cmd, conf,
381                                 ticker, ticker._pattern);
382                         break;
383                     case "wait":
384                         ticker.waitTo = this.age + eval(cmd.value);
385                         return;
386                     case "changeDirection":
387                         ticker._pattern._changeDirection.call(this,
388                                 cmd, conf, ticker);
389                         break;
390                     case "changeSpeed":
391                         ticker._pattern._changeSpeed.call(this, cmd,
392                                 ticker);
393                         break;
394                     case "accel":
395                         ticker._pattern._accel.call(this, cmd, ticker);
396                         break;
397                     case "vanish":
398                         if (this.parentNode) {
399                             this.parentNode.removeChild(this);
400                             // tickerPool.dispose(ticker);
401                         }
402                         break;
403                     }
404                 }
405 
406                 // complete
407                 ticker.completed = true;
408                 if (ticker.parentTicker) {
409                     ticker.parentTicker.completeChild();
410                 } else {
411                     this.dispatchEvent(new Event("completeAttack"));
412                 }
413             };
414 
415             if (action === undefined) {
416                 ticker.walker = this._bulletml.getWalker("top",
417                         config.rank);
418             } else if (typeof (action) === "string") {
419                 ticker.walker = this._bulletml.getWalker(action,
420                         config.rank);
421             } else if (action instanceof BulletML.Bullet) {
422                 ticker.walker = action.getWalker(config.rank);
423             } else {
424                 console.error(config, action);
425                 throw new Error("引数が不正");
426             }
427 
428             ticker._pattern = this;
429             ticker.config = config;
430             ticker.waitTo = -1;
431             ticker.completed = false;
432             ticker.direction = 0;
433             ticker.lastDirection = 0;
434             ticker.speed = 0;
435             ticker.lastSpeed = 0;
436             ticker.speedH = 0;
437             ticker.speedV = 0;
438             ticker.dirIncr = 0;
439             ticker.dirFin = 0;
440             ticker.chDirEnd = -1;
441             ticker.spdIncr = 0;
442             ticker.spdFin = 0;
443             ticker.chSpdEnd = -1;
444             ticker.aclIncrH = 0;
445             ticker.aclFinH = 0;
446             ticker.aclIncrV = 0;
447             ticker.aclFinV = 0;
448             ticker.aclEnd = -1;
449 
450             ticker.isDanmaku = true;
451             return ticker;
452         },
453         _fire : function(cmd, config, ticker, pattern) {
454             var b = config.bulletFactory({
455                 label : cmd.bullet.label
456             });
457             if (!b) {
458                 return;
459             }
460 
461             var bt = pattern.createTicker(config, cmd.bullet);
462 
463             var attacker = this;
464             var calcDirection = function(d) {
465                 var dv = toRadian(eval(d.value));
466                 // console.debug(d.type);
467                 switch (d.type) {
468                 case "aim":
469                     if (config.target) {
470                         return radiusAtoB(attacker, config.target) + dv;
471                     } else {
472                         return dv - Math.PI / 2;
473                     }
474                 case "absolute":
475                     return dv - Math.PI / 2; // 真上が0度
476                 case "relative":
477                     return ticker.direction + dv;
478                 case "sequence":
479                     // console.debug(ticker.lastDirection, dv);
480                     return ticker.lastDirection + dv;
481                 }
482             };
483             bt.direction = calcDirection(cmd.direction
484                     || cmd.bullet.direction);
485             ticker.lastDirection = bt.direction;
486             // console.debug(bt.direction);
487 
488             var calcSpeed = function(s) {
489                 var sv = eval(s.value);
490                 switch (s.type) {
491                 case "relative":
492                 case "sequence":
493                     return ticker.lastSpeed + sv;
494                 case "absolute":
495                 default:
496                     return sv;
497                 }
498             };
499             bt.speed = calcSpeed(cmd.speed || cmd.bullet.speed);
500             ticker.lastSpeed = bt.speed;
501 
502             b.x = this.x + ((this.width || 0) - (b.width || 0)) / 2;
503             b.y = this.y + ((this.height || 0) - (b.height || 0)) / 2;
504 
505             b.addEventListener("enterframe", bt);
506             b.addEventListener("removed", function() {
507                 this.removeEventListener("enterframe", bt);
508             });
509             if (config.addTarget) {
510                 config.addTarget.addChild(b);
511             } else if (this.parentNode) {
512                 this.parentNode.addChild(b);
513             }
514         },
515         _changeDirection : function(cmd, config, ticker) {
516             var d = eval(cmd.direction.value);
517             var t = eval(cmd.term);
518             switch (cmd.direction.type) {
519             case "aim":
520                 var tar = config.target;
521                 if (!tar) {
522                     return;
523                 }
524                 ticker.dirFin = radiusAtoB(this, tar) + toRadian(d);
525                 ticker.dirIncr = rel(ticker.dirFin - ticker.direction) / t;
526                 break;
527             case "absolute":
528                 ticker.dirFin = toRadian(d) - Math.PI / 2;
529                 ticker.dirIncr = rel(ticker.dirFin - ticker.direction) / t;
530                 break;
531             case "relative":
532                 ticker.dirFin = ticker.direction + toRadian(d);
533                 ticker.dirIncr = rel(ticker.dirFin - ticker.direction) / t;
534                 break;
535             case "sequence":
536                 ticker.dirIncr = toRadian(d);
537                 ticker.dirFin = ticker.direction + ticker.dirIncr * t;
538                 break;
539             }
540             ticker.chDirEnd = this.age + t;
541         },
542         _changeSpeed : function(cmd, ticker) {
543             var s = eval(cmd.speed.value);
544             var t = eval(cmd.term);
545             switch (cmd.speed.type) {
546             case "absolute":
547                 ticker.spdFin = s;
548                 ticker.spdIncr = (ticker.spdFin - ticker.speed) / t;
549                 break;
550             case "relative":
551                 ticker.spdFin = s + ticker.speed;
552                 ticker.spdIncr = (ticker.spdFin - ticker.speed) / t;
553                 break;
554             case "sequence":
555                 ticker.spdIncr = s;
556                 ticker.spdFin = ticker.speed + ticker.spdIncr * t;
557                 break;
558             }
559             ticker.chSpdEnd = this.age + t;
560         },
561         _accel : function(cmd, ticker) {
562             var t = eval(cmd.term);
563             ticker.aclEnd = this.age + t;
564 
565             if (cmd.horizontal) {
566                 var h = eval(cmd.horizontal.value);
567                 switch (cmd.horizontal.type) {
568                 case "absolute":
569                 case "sequence":
570                     ticker.aclIncrH = (h - ticker.speedH) / t;
571                     ticker.aclFinH = h;
572                     break;
573                 case "relative":
574                     ticker.aclIncrH = h;
575                     ticker.aclFinH = (h - ticker.speedH) * t;
576                     break;
577                 }
578             } else {
579                 ticker.aclIncrH = 0;
580                 ticker.aclFinH = ticker.speedH;
581             }
582 
583             if (cmd.vertical) {
584                 var v = eval(cmd.vertical.value);
585                 switch (cmd.vertical.type) {
586                 case "absolute":
587                 case "sequence":
588                     ticker.aclIncrV = (v - ticker.speedV) / t;
589                     ticker.aclFinV = v;
590                     break;
591                 case "relative":
592                     ticker.aclIncrV = v;
593                     ticker.aclFinV = (v - ticker.speedV) * t;
594                     break;
595                 }
596             } else {
597                 ticker.aclIncrV = 0;
598                 ticker.aclFinV = ticker.speedV;
599             }
600         },

629     /**
630      * ラジアンから度数に変換.
631      */
632     function toDegree(radian) {
633         return radian * 180 / Math.PI;
634     }
635     /**
636      * 度数からラジアンに変換.
637      */
638     function toRadian(degree) {
639         return degree * Math.PI / 180;
640     }
641     /**
642      * ラジアンを -π<= rad < π の範囲に正規化する.
643      */
644     function rel(radian) {
645         while (radian <= -Math.PI) {
646             radian += Math.PI * 2;
647         }
648         while (Math.PI < radian) {
649             radian -= Math.PI * 2;
650         }
651         return radian;
652     }
653     /**
654      * スプライトAから見たスプライトBの方向をラジアンで返す.
655      * 
656      * @param {enchant.Node}
657      *            a スプライトA
658      * @param {enchant.Node}
659      *            b スプライトB
660      */
661     function radiusAtoB(a, b) {
662         var ca = {
663             x : a.x + (a.width || 0) / 2,
664             y : a.y + (a.height || 0) / 2
665         };
666         var cb = {
667             x : b.x + (b.width || 0) / 2,
668             y : b.y + (b.height || 0) / 2
669         };
670         return Math.atan2(cb.y - ca.y, cb.x - ca.x);
671     }
